    # 이미 계산한적이 있는 문제면 그대로 반환

    if x==1 or x==2:

    d[x] = fibo(x-1)+fibo(x-2) # fibo(5) = fibo(4) + fibo(3)       fibo(3) + fibo(2)    fibo(2) + fibo(1)
